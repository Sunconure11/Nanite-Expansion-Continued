	}

#Subsumed Armies: Subsuming the enemy's army as your own
#An army has been killed in ground combat
# This = owner
# From = army
# FromFrom = opponent
# FromFromFrom = planet
country_event = {
	id = nano.1350
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		exists = fromfrom
		fromfrom = {
			any_owned_army = {
				army_type = nanoblight_army
			}
		}
		from = {
			exists = species
			species = {
				is_robotic = no
			}
		}
		NOT = { is_country_type = "gate_builders" }
	}

	immediate = {
		random_list = {
			2 = { }
			1 = {
				if = {
					limit = {
						fromfromfrom = {
							OR = {
								any_ground_combat_defender = {
									army_type = nanoblight_army
								}
								any_ground_combat_attacker = {
									army_type = nanoblight_army
								}
							}
						}
						fromfrom = {
							any_owned_army = {
								army_type = nanoblight_army
							}
						}
					}
					fromfromfrom = {
						create_army = {
							owner = root.fromfrom
							type = nanite_subsumed_army
							species = root.from.species
						}
					}
				}
			}
		}
	}
}
