nanoroboticist = {
	category = specialist

	swappable_data = {
		default = {
			condition_string = SPECIALIST_JOB_TRIGGER
			building_icon = building_nanite_fabricator_1
		}
	}

	# clothes_texture_index = 3
	tags = { nanites nectd_job_tag_nanites }

	possible_pre_triggers = {
		has_owner = yes
		is_sapient = yes
		is_being_assimilated = no
		is_being_purged = no
	}

	possible_precalc = can_fill_specialist_job

	resources = {
		category = planet_nanoroboticists
		produces = {
			nanites = 2
		}
		produces = {
			trigger = {
				owner = { has_technology = tech_attoscale_mechanics }
			}
			nanites = 1
		}
		upkeep = {
			alloys = 5
		}
		upkeep = {
			trigger = {
				owner = { has_technology = tech_attoscale_mechanics }
			}
			alloys = 2
		}
	}

	weight = {
		weight = @specialist_job_weight
		modifier = {
			factor = 0.2
			NOR = { has_citizenship_rights = yes has_trait = trait_mechanical }
		}
		modifier = {
			factor = 0.1
			can_take_servant_job = yes
		}
		modifier = {
			factor = 2
			species = {
				exists = event_target:racket_species is_same_species = event_target:racket_species
			}
		}
		modifier = {
			factor = 7
			any_owned_pop_job = {
				has_job_type = nanoroboticist
			}
		}
		modifier = {
			factor = 15
			any_owned_pop_job = {
				has_job_type = nanoroboticist
			}
			OR = {
				has_trait = trait_robust
				has_trait = trait_robot_efficient_processors
			}
		}
		modifier = {
			factor = 1.2
			is_enslaved = yes
			has_slavery_type = { type = slavery_indentured }
		}
		modifier = {
			factor = 2
			has_trait = trait_void_dweller_1
		}
		modifier = {
			factor = 1.5
			has_trait = trait_necrophage
		}
	}
}
