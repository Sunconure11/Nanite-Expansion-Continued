building_nanite_fabricator_1 = {
	base_buildtime = @b3_time

	category = manufacturing

	destroy_trigger = {
		OR = {
			has_modifier = resort_colony
			has_modifier = slave_colony
			AND = {
				exists = owner
				buildings_no_motes = yes
				buildings_no_crystals = yes
				buildings_no_exotic_gases = yes
			}
		}
	}

	allow = {
		has_upgraded_capital = yes
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = no }
		}
		modifier = {
			job_nanoroboticist_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		modifier = {
			job_nanoroboticist_drone_add = 1
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			alloys = 1000
		}
		upkeep = {
			rare_crystals = 2
			volatile_motes = 2
			exotic_gases = 2
			energy = 5
		}
	}

	upgrades = {
		building_nanite_fabricator_2
	}

	prerequisites = {
		"tech_micromechanic_constructs"
	}

	ai_resource_production = {
		nanites = 2
		trigger = {
			always = yes
		}
	}
}

building_universal_constructor_1 = {
	base_buildtime = @b1_time
	base_cap_amount = 1
	category = manufacturing

	potential = {
		NOR = {
			has_modifier = resort_colony
			has_modifier = slave_colony
		}
	}

	allow = {
		has_upgraded_capital = yes
	}

	destroy_trigger = {
		OR = {
			has_modifier = resort_colony
			has_modifier = slave_colony
		}
	}

	resources = {
		category = planet_buildings

		# Normal Planets Tier 1
        produces = {
			alloys = 2
			consumer_goods = 2
			nanites = -1
			mult = planet.value:toxic_attendant_growth
		}
		produces = { 
			trigger = { 
				owner = {
					has_technology = tech_atto_optimizations
				}
			} 
			alloys = 2
			consumer_goods = 2
			nanites = -1
			mult = planet.value:toxic_attendant_growth
		}

		cost = {
			alloys = 1000
			nanites = @b2_rare_cost
		}
		upkeep = {
			energy = @b2_upkeep
			nanites = 5
		}
	}

	upgrades = {
		building_universal_constructor_2
	}

	prerequisites = {
		tech_self_reconfiguring_micromatter
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { NOT = { has_technology = tech_atto_optimizations } }
			NOT = {
				is_planet_class = pc_ringworld_habitable
			}
		}
		text = building_universal_constructor_1_modifier_desc
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { has_technology = tech_atto_optimizations }
			NOT = {
				is_planet_class = pc_ringworld_habitable
			}
		}
		text = building_universal_constructor_1_atto_modifier_desc
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { NOT = { has_technology = tech_atto_optimizations } }
			is_planet_class = pc_ringworld_habitable
		}
		text = building_universal_constructor_1_ring_modifier_desc
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { has_technology = tech_atto_optimizations }
			is_planet_class = pc_ringworld_habitable
		}
		text = building_universal_constructor_1_ring_atto_modifier_desc
	}
}
