building_nanite_fabricator_1 = {
	base_buildtime = @b3_time
	category = manufacturing

	building_sets = {
		nec_building_set_all
		nec_building_set_assembly
		industrial
		foundry
	}

	potential = {
		has_owner = yes
		NOR = {
			has_modifier = resort_colony
			has_modifier = slave_colony
		}
		OR = {
			owner = { is_ai = no }
			nectd_has_any_generator_district_or_building = yes
		}
	}

	allow = {
		is_occupied_flag = no
		has_upgraded_capital = yes
	}

	destroy_trigger = {
		OR = {
			has_owner = no
			has_modifier = resort_colony
			has_modifier = slave_colony
			AND = {
				exists = owner
				owner = { is_ai = yes }
				nectd_has_any_generator_district_or_building = no
			}
		}
	}

	triggered_planet_modifier = {
		potential = {
			has_owner = yes
			owner = { is_gestalt = no }
		}
		job_nanoroboticist_add = 100
	}
	triggered_planet_modifier = {
		potential = {
			has_owner = yes
			owner = { is_gestalt = yes }
		}
		job_nanoroboticist_drone_add = 100
	}

	resources = {
		category = planet_buildings
		cost = {
			alloys = 1000
		}
		upkeep = {
			rare_crystals = 2
			volatile_motes = 2
			exotic_gases = 2
			energy = 5
		}
	}

	upgrades = {
		building_nanite_fabricator_2
	}

	prerequisites = {
		"tech_micromechanic_constructs"
	}

	ai_resource_production = {
		nanites = 2
		trigger = { always = yes }
	}

}

building_universal_constructor_1 = {
	base_buildtime = @b1_time
	base_cap_amount = 1
	category = manufacturing

	building_sets = {
		nec_building_set_all
		nec_building_set_production
		nec_building_set_foundry
		nec_building_set_factory
		industrial
		foundry
	}

	potential = {
		NOR = {
			has_modifier = resort_colony
			has_modifier = slave_colony
		}
	}

	allow = {
		is_occupied_flag = no
		has_upgraded_capital = yes
	}

	destroy_trigger = {
		OR = {
			has_modifier = resort_colony
			has_modifier = slave_colony
		}
	}

	resources = {
		category = planet_buildings
		produces = {
			trigger = {
				exists = owner
				owner = {
					NOT = { has_technology = tech_atto_optimizations }
				}
			}
			alloys = 3  # was 2
			consumer_goods = 3  # was 2
			# nanites = -1
			mult = value:nectd_calc_factory_foundry_jobs_div_100
		}
		produces = {
			trigger = {
				exists = owner
				owner = { has_technology = tech_atto_optimizations }
			}
			alloys = 2
			consumer_goods = 2
			# nanites = -1
			mult = value:nectd_calc_factory_foundry_jobs_div_100
			mult = 2
		}
		cost = {
			alloys = 1000
			nanites = @b2_rare_cost
		}
		upkeep = {
			energy = @b2_upkeep
			nanites = 5
		}
		# NANITE UPKEEP
		# Takes former negative nanite production into upkeep
		upkeep = {
			trigger = {
				exists = owner
				owner = {
					NOT = { has_technology = tech_atto_optimizations }
				}
			}
			nanites = 1
			mult = value:nectd_calc_factory_foundry_jobs_div_100
		}

		upkeep = {
			trigger = {
				exists = owner
				owner = { has_technology = tech_atto_optimizations }
			}
			nanites = 1
			mult = value:nectd_calc_factory_foundry_jobs_div_100
			mult = 2
		}
	}

	upgrades = {
		building_universal_constructor_2
	}

	prerequisites = {
		tech_self_reconfiguring_micromatter
	}

	triggered_desc = {
		trigger = {
			has_owner = yes
			NOR = {
				owner = { has_technology = tech_atto_optimizations }
				has_ringworld_output_boost = yes
			}
		}
		text = building_universal_constructor_1_modifier_desc
	}
	triggered_desc = {
		trigger = {
			has_owner = yes
			owner = { has_technology = tech_atto_optimizations }
			has_ringworld_output_boost = no
		}
		text = building_universal_constructor_1_atto_modifier_desc
	}
	triggered_desc = {
		trigger = {
			has_owner = yes
			owner = {
				NOT = { has_technology = tech_atto_optimizations }
			}
			has_ringworld_output_boost = yes
		}
		text = building_universal_constructor_1_ring_modifier_desc
	}
	triggered_desc = {
		trigger = {
			has_owner = yes
			owner = { has_technology = tech_atto_optimizations }
			has_ringworld_output_boost = yes
		}
		text = building_universal_constructor_1_ring_atto_modifier_desc
	}
}
